package edu.brown.cs.student.main;

import edu.brown.cs.student.main.parsing.Parser;
import edu.brown.cs.student.main.parsing.creators.StringCreator;
import edu.brown.cs.student.main.searching.Search;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;

/** The Main class of our project. This is where execution begins. */
public final class Main {
  /**
   * The initial method called when execution begins.
   *
   * @param args An array of command line arguments
   */
  public static void main(String[] args) throws IOException {
    new Main(args).run();
  }

  /**
   * User input and primary logic is all conducted here. In essence the file reader, creator,
   * parser, and searcher are all instantiated here. Arguments that the user inputs are used to fill
   * in the file being used, target value that will be needed, and what indicators are being used by
   * the user to narrow the search.
   *
   * @param args
   * @throws IOException
   */
  private Main(String[] args) throws IOException {
    String filename = args[0]; // file path is stored in a string
    FileReader fileReader = null; // filereader initialized
    try {
      fileReader =
          new FileReader(filename); // this try catch makes sure a valid file path is inputted
    } catch (FileNotFoundException e) {
      System.out.println("Cannot find file " + filename);
    }
    StringCreator creator = new StringCreator();
    Parser parser = new Parser<>(creator, fileReader);
    Search searcher = new Search(parser, args[1]);
    if (args.length > 2) { // this logic checks to see if an indicator has been inputted by the user
      if (args[2].equalsIgnoreCase("true")) { // if user inputs true, a header is being inputted
        System.out.println(searcher.findTargetHeader(args[3]));
      } else if ((args.length == 4)
          && args[2].equalsIgnoreCase(
              "false")) { // in this case it is known that a header is not coming, but
        System.out.println(
            searcher.findTargetColumn(
                Integer.parseInt(args[3]))); // turn the string arg into an int
      }
      System.out.println(
          searcher
              .findTargetAll()); // if the user messes up the arguments, just conduct the default
      // search
    } else {
      System.out.println(searcher.findTargetAll()); // else run the default case
    }
  }

  private void run() {}
}
