package edu.brown.cs.student.main.parsing.creators;

import edu.brown.cs.student.main.parsing.CreatorFromRow;
import edu.brown.cs.student.main.parsing.Person;
import java.util.List;

/**
 * This is the creator class that is testing the implementation of CreatorFromRow in a custom Object
 * class, simulating what a developer who wants to use the Parse software uniquely might want to
 * engage in.
 */
public class PersonCreator implements CreatorFromRow<Person> {

  /**
   * Create method that turns the List of Strings that the split CSV rows are outputed in into a
   * custom class type of Person.
   */
  public PersonCreator() {}

  /**
   * The only method is the create method implemented from the interface CreatorFromRow, which takes
   * in a row of Strings, and for that row indexes into the name column, age column, and gender
   * column to output a type Person with those three fields.
   *
   * @param row
   * @return
   */
  @Override
  public Person create(List<String> row) {
    String name = row.get(0);
    String age = row.get(1);
    String gender = row.get(2);
    Person person = new Person(name, age, gender);
    return person;
  }
}
