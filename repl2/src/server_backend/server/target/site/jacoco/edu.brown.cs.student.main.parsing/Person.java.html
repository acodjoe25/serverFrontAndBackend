package edu.brown.cs.student.main.parsing;

/**
 * Custom class to demo how a developer could use a unique creator to parse the CSV rows. This is an
 * example class of a Person
 */
public class Person {

  private String name;
  private String age;
  private String gender;

  /**
   * The constructor takes in three fields, all are Strings being the name, age, and gender.
   *
   * @param name
   * @param age
   * @param gender
   */
  public Person(String name, String age, String gender) {
    this.name = name;
    this.age = age;
    this.gender = gender;
  }

  /**
   * Getter method that returns the name.
   *
   * @return
   */
  public String getName() {
    return this.name;
  }

  /**
   * Getter method that returns the age.
   *
   * @return
   */
  public String getAge() {
    return this.age;
  }

  /**
   * Getter method that returns the gender.
   *
   * @return
   */
  public String getGender() {
    return this.gender;
  }
}
